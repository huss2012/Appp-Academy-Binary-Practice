## Binary to base 10
1. 0b1010 = 10
2. 0b0011 = 3

| Binary      | Base 10     |
| ----------- | ----------- |
| 1. 0b1010   | 10          |
| 2. 0b0011   | 3           |
## Binary to hexadecimal
1. 0b1010 = 0xA
2. 0b0011 = 0x3

| Binary      | Hex         |
| ----------- | ----------- |
| 1. 0b1010   | 0xA         |
| 2. 0b0011   | 0X3         |
## Hexadecimal to base 10
1. 0xa1 = 161
2. 0xff = 255

| Hex         | Base 10     |
| ----------- | ----------- |
| 1. 0xa1     | 161         |
| 2. 0xff     | 255         |
## Hexadecimal to binary
1. 0xa1 = 10100001
2. 0xff = 11111111

| Hex         | Binary      |
| ----------- | ----------- |
| 1. 0xa1     | 10100001    |
| 2. 0xff     | 11111111    |
## Base 10 to binary##
1. 8    = 00001000
2. 24   = 00011000
3. 255  = 11111111

| Base 10     | Binary      |
| ----------- | ----------- |
| 1. 8        | 00001000    |
| 2. 24       | 00011000    |
| 3. 255      | 11111111    |
## Base 10 to hexadecimal
1. 8    = 0x8

3. 255  = 0xff

| Base 10     | Hex         |
| ----------- | ----------- |
| 1. 8        | 0x8         |
| 2. 24       | 0x18        |
| 3. 255      | 0xff        |
